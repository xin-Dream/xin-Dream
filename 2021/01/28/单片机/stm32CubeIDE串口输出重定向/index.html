重定向函数

在配置好串口之后需要在相关文件中添加以下函数以实现printf函数的重定向

//注意添加头文件
#include "stdio.h"

#ifdef __GNUC__
#define PUTCHAR_PROTOTYPE int __io_putchar(int ch)
#else
#define PUTCHAR_PROTOTYPE int fputc(int ch, FILE *f)
#endif

PUTCHAR_PROTOTYPE
{
    HAL_UART_Transmit(&huart1, (uint8_t*) &ch, 1, 0xffff);
    return ch;
}
